import MinoDiv from "components/MinoDiv"
import Polyomino from "mino/Polyomino"

A polyomino that is "fixed" in place; that is, any rotation or reflection counts as a different polyomino.

For example, the following transformations of the L tetromino count as different fixed polyominoes, even though they count as the same [free](/glossary/free) polyomino.

<Example />

export function Example() {
  const mino = Polyomino.of("100_111")
  return (
    <div style={{ display: "flex", gap: "1rem", alignItems: "center" }}>
      {mino.transform.all().map((transform) => (
        <MinoDiv mino={transform} stroke="black" fill="grey" size={15} />
      ))}
    </div>
  )
}
